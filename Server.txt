
import java.io.*;
import java.net.*;
import java.util.*;

public class TicTacToeClient {

    public static void main(String args[]) throws Exception {
        Socket clientSocket = new Socket("150.254.78.69", 8787);
        DataOutputStream outToServer = new DataOutputStream(clientSocket.getOutputStream());
        BufferedReader inFromServer = new BufferedReader(new InputStreamReader(clientSocket.getInputStream()));
        int table[] = new int[25];
        String opponent;
        int opponentMove;
        int myMove;
        int gamesWon = 0;
        inFromServer.readLine();
        outToServer.writeBytes("LOGIN julia_may");
        for (int i = 0; i < 200; i++) {
            Arrays.fill(table, 0);
            myMove = 0;
            outToServer.writeBytes("MOVE " + Integer.toString(myMove));
            table[myMove] = 1;
            myMove++;
            opponent = inFromServer.readLine();
            while (opponent != "WIN" && opponent != "LOST") {
                opponentMove = Integer.parseInt(opponent.split(" ")[1]);
                table[opponentMove] = 2;
                while (table[myMove] == 2) {
                    myMove++;
                }
                outToServer.writeBytes("MOVE " + Integer.toString(myMove));
                table[myMove] = 1;
                myMove++;
                opponent = inFromServer.readLine();
            }
            if (opponent == "WIN") {
                gamesWon++;
            }
            inFromServer.readLine();
        }
        System.out.println(inFromServer.readLine());
        System.out.println(gamesWon);
        clientSocket.close();
    }

}
